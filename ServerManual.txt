## 1. Compose Docker Database
##  install docker https://docs.docker.com/get-started/get-docker/
##  install docker-compose https://docs.docker.com/compose/install/
##  navigate to IDERHA directory
docker-compose up -d

## Connect to docker using shell if needed

docker exec -it iderha_postgres psql -U postgres -d IDERHA


## 2. Start Backend Server ##

cd backend
## Check maven version if installed
mvn -v
mvn clean install
mvn spring-boot:run



## 3. Start cd s Server ##

cd react-app
## Check node.js and Node Package Manager if installed
node -v
npm -v
npm run dev

## Frontend build fix
## cd react-app
## npm install
## npm install -g vite



## 4. Testing Frontend

## Install vitest and jsdom packages for testing:
cd react-app

npm install jsdom --save-dev

cd react-app
